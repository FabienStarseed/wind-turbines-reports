# Plan 05-02 — index.html Layout + Header + Tab Navigation

## Goal

Write the structural shell of `frontend/index.html` with Cyber Operator design system applied: fixed header with AWID branding, tab navigation (NEW INSPECTION / HISTORY), and the Alpine.js `app()` component skeleton. Form fields and history table are wired in Plans 05-03 and 05-04 — this plan creates the correct outer structure both depend on.

## must_haves

- Fixed 64px header with `position: fixed; z-index: 100` — content area has `padding-top: 64px`
- Header left: `AWID` (Bebas Neue, cyan gradient) + `APAC WIND INSPECTIONS DRONES` subtitle (muted mono, xs)
- Header center: tab nav — `NEW INSPECTION` and `HISTORY` buttons — amber bottom border on active tab
- Header right: username pill (JetBrains Mono, cyan) + `LOGOUT` action-btn (amber variant)
- Tab switching via Alpine `activeTab` state (`'new-inspection'` | `'history'`), content toggled with `x-show`
- Page body uses `bg-[#050510]` + animated background (handled by style.css — body::before, body::after, scanlines div)
- All CDN imports present: Tailwind Play CDN, tailwind.config override, Google Fonts, Alpine.js v3
- Link to `style.css`
- Alpine `app()` function declared in inline `<script>` — skeleton with all state properties and method stubs (implementations come in 05-03 and 05-04)
- `x-cloak` on body, `[x-cloak] { display:none }` covered in style.css
- Unauthenticated users redirected to `/login` immediately in `init()`
- Tablet responsive: header tabs remain on one row at 768px (do NOT stack)

    ```js
    function app() {
      return {
        // ── Auth ──
        token: localStorage.getItem('bdda_token') || null,
        username: localStorage.getItem('bdda_username') || '',

        // ── Tab navigation ──
        activeTab: 'new-inspection',

        // ── Form sections collapse state ──
        sections: {
          identity: true,
          details: true,
          environment: true,
          notes: true,
          upload: true,
        },

        // ── File upload state ──
        fileSelected: false,
        fileName: '',
        fileSize: '',

        // ── Pipeline / job state ──
        jobId: null,
        stage: null,
        progress: 0,
        message: '',
        totalImages: null,
        flaggedImages: null,
        criticalFindings: null,
        isPolling: false,
        downloadAvailable: false,
        autoRedirect: false,
        countdown: 3,
        pipelineError: null,
        submitting: false,
        _pollTimer: null,

        // ── Stage definitions (updated — no Kimi/Gemini) ──
        stages: [
          { key: 'ingesting',         label: 'INGEST IMAGES' },
          { key: 'triaging',          label: 'TRIAGE — claude-opus-4-6' },
          { key: 'classifying',       label: 'CLASSIFY — claude-opus-4-6' },
          { key: 'analyzing',         label: 'ANALYZE — claude-opus-4-6' },
          { key: 'generating_report', label: 'BUILD PDF' },
          { key: 'complete',          label: 'COMPLETE' },
        ],
        STAGE_ORDER: ['ingesting','triaging','classifying','analyzing','generating_report','complete'],

        // ── History ──
        jobs: [],
        historyLoaded: false,

        // ── Init ──
        async init() {
          if (!this.token) { window.location.href = '/login'; return; }
          // Decode username from token if not stored
          if (!this.username) {
            try {
              const p = JSON.parse(atob(this.token.split('.')[1]));
              this.username = p.sub || '';
              localStorage.setItem('bdda_username', this.username);
            } catch {}
          }
        },

        // ── Auth ──
        logout() {
          localStorage.removeItem('bdda_token');
          localStorage.removeItem('bdda_username');
          window.location.href = '/login';
        },

        // ── API wrapper (adds Bearer, handles 401, silent refresh) ──
        async apiFetch(path, opts = {}) {
          const token = localStorage.getItem('bdda_token');
          if (!token) { this.logout(); return null; }
          const headers = { 'Authorization': `Bearer ${token}`, ...(opts.headers || {}) };
          const res = await fetch(path, { ...opts, headers });
          const newToken = res.headers.get('X-New-Token');
          if (newToken) { localStorage.setItem('bdda_token', newToken); this.token = newToken; }
          if (res.status === 401) { this.logout(); return null; }
          return res;
        },

        // ── Stage helpers ──
        stageIndex() { return this.STAGE_ORDER.indexOf(this.stage); },
        isDone(i)    { return this.stageIndex() > i && this.stage !== 'error'; },
        isActive(i)  { return this.stageIndex() === i && this.stage !== 'error'; },
        isFailed(i)  { return this.stage === 'error' && this.stageIndex() === i; },
        isPending(i) { return this.stageIndex() < i; },

        // ── Date formatter ──
        formatDate(iso) {
          if (!iso) return '—';
          const d = new Date(iso);
          const mon = d.toLocaleDateString('en-GB', { month: 'short' }).toUpperCase();
          const day = String(d.getDate()).padStart(2, '0');
          const time = d.toLocaleTimeString('en-GB', { hour: '2-digit', minute: '2-digit' });
          return `${mon} ${day} · ${time}`;
        },

        // ── Methods stubs (implemented in 05-03 and 05-04) ──
        handleFileChange(e) {},
        handleDrop(e) {},
        async submit() {},
        startPolling(jobId) {},
        stopPolling() {},
        async pollOnce(jobId) {},
        async downloadReport(jobId) {},
        async loadHistory() {},
      }
    }
    ```

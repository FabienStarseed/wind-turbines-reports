# Plan 05-03 — Upload Form + Alpine State + Live Progress Tracker

## Goal

Populate the "NEW INSPECTION" tab in `index.html` with:
1. The full upload form (all 5 collapsible sections, Cyber Operator styling, file drop zone)
2. The live pipeline progress tracker (stage dots, progress bar, celebration state, error state)
3. All Alpine methods: `submit()`, `startPolling()`, `pollOnce()`, `handleFileChange()`, `handleDrop()`, `downloadReport()`

This plan edits `frontend/index.html` — the shell from Plan 05-02 must exist first.

## must_haves

- All 5 form sections present and collapsible (Alpine `x-show` + `x-transition`, default expanded)
- Section headers use cyan line + uppercase JetBrains Mono pattern with `▼`/`▶` toggle icon
- Form grid: 2-column on desktop, collapses to 1-column below 768px (`grid-cols-1 sm:grid-cols-2`)
- Drop zone: dashed border turns green + shows filename when file selected
- Submit button: `action-btn` clip-path, `▶ GENERATE INSPECTION REPORT`, disabled during pipeline
- `submit()` uses `apiFetch('/api/upload', { method:'POST', body: formData })` — Bearer token via apiFetch wrapper (fixes missing auth bug)
- Progress tracker shows only after first submit (`x-if` gates the whole panel, not `x-show`)
- 6 stage dots rendered via `x-for` over `stages` array — pending/active/done/failed states
- Progress bar: 4px thin, amber→cyan gradient fill, turns fully green on complete
- On `stage === 'complete'`: "INSPECTION COMPLETE" badge (green, pulsing), download button, auto-redirect checkbox
- `downloadReport()` uses blob fetch pattern — NOT direct `<a href>` (fixes JWT auth bug on download)
- Auto-redirect: if checkbox checked, countdown from 3 then `activeTab = 'history'` + `loadHistory()`
- Error state: magenta stage dot + magenta error message below progress bar
- Polling: 2s interval via `setInterval`, stored on `this._pollTimer`, stopped on complete/error
- Inspection date field defaults to today's date in `init()` (after existing init code)

    handleFileChange(e) {
      const file = e.target.files[0];
      if (!file) return;
      this.fileSelected = true;
      this.fileName = file.name;
      this.fileSize = (file.size / 1024 / 1024).toFixed(1) + ' MB';

    `handleDrop(e)`:
    ```js
    handleDrop(e) {
      const file = e.dataTransfer.files[0];
      if (!file) return;
      // Assign to actual file input
      const input = document.getElementById('fileInput');
      const dt = new DataTransfer();
      dt.items.add(file);
      input.files = dt.files;
      this.fileSelected = true;
      this.fileName = file.name;
      this.fileSize = (file.size / 1024 / 1024).toFixed(1) + ' MB';
    },
    ```

    `submit()`:
    ```js
    async submit() {
      const fileInput = document.getElementById('fileInput');
      if (!fileInput.files.length) return;

      this.submitting = true;
      this.jobId = null;
      this.stage = null;
      this.progress = 0;
      this.message = '';
      this.pipelineError = null;
      this.downloadAvailable = false;

      const formData = new FormData(document.querySelector('form'));
      if (!formData.get('inspection_date')) {
        formData.set('inspection_date', new Date().toISOString().split('T')[0]);
      }

      try {
        const res = await this.apiFetch('/api/upload', { method: 'POST', body: formData });
        if (!res || !res.ok) {
          const err = await res?.json().catch(() => ({ detail: 'Upload failed' }));
          throw new Error(err.detail || 'Upload failed');
        }
        const data = await res.json();
        this.jobId = data.job_id;
        this.totalImages = data.image_count || null;
        this.startPolling(data.job_id);
      } catch (err) {
        this.pipelineError = err.message;
        this.stage = 'error';
      } finally {
        this.submitting = false;
      }
    },
    ```

    `startPolling(jobId)`:
    ```js
    startPolling(jobId) {
      this.stopPolling();
      this.isPolling = true;
      this._pollTimer = setInterval(() => this.pollOnce(jobId), 2000);
      this.pollOnce(jobId);
    },
    ```

    `stopPolling()`:
    ```js
    stopPolling() {
      if (this._pollTimer) { clearInterval(this._pollTimer); this._pollTimer = null; }
      this.isPolling = false;
    },
    ```

    `pollOnce(jobId)`:
    ```js
    async pollOnce(jobId) {
      try {
        const res = await this.apiFetch(`/api/status/${jobId}`);
        if (!res || !res.ok) return;
        const s = await res.json();
        this.stage = s.stage;
        this.progress = s.progress >= 0 ? s.progress : 0;
        this.message = s.message || '';
        this.totalImages = s.total_images || this.totalImages;
        this.flaggedImages = s.flagged_images || null;
        this.criticalFindings = s.critical_findings || null;

        if (s.stage === 'complete') {
          this.stopPolling();
          this.downloadAvailable = true;
          this.progress = 100;
          this.loadHistory();
          // Auto-redirect countdown
          if (this.autoRedirect) {
            this.countdown = 3;
            const tick = setInterval(() => {
              this.countdown--;
              if (this.countdown <= 0) {
                clearInterval(tick);
                this.activeTab = 'history';
              }
            }, 1000);
          }
        }
        if (s.stage === 'error' || s.stage === 'failed' || s.stage === 'cost_limit_exceeded') {
          this.stage = 'error';
          this.pipelineError = s.error || s.message || 'Pipeline failed.';
          this.stopPolling();
        }
      } catch {
        // Network error — keep polling
      }
    },
    ```

    `downloadReport(jobId)`:
    ```js
    async downloadReport(jobId) {
      const res = await this.apiFetch(`/api/download/${jobId}`);
      if (!res || !res.ok) return;
      const blob = await res.blob();
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `inspection_${jobId}.pdf`;
      a.click();
      URL.revokeObjectURL(url);
    },
    ```

    }
